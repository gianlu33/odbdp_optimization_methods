GIT ISTRUZIONI

git log:
	visualizza elenco commit
	
	git log --graph --oneline branch1 branch2 ...
		visualizza log di più branches, con ogni commit su ogni riga (--oneline) e a grafo,
		per dividere le varie branches (--oneline)
	
	git log -n 2
		visualizza ultimi 2 commit
		
git checkout xxx:
	ritorna a un commit precedente, quello con id xxx
	
	git checkout nomebranch
		passa a un altro branch
		
	git checkout -b nomebranch	
		crea branch dal punto in cui ho fatto checkout

git diff:
	visualizza differenze
	
	git diff xxx yyy 
		visualizza differenze tra commit xxx e yyy (yyy sarebbe il nuovo)
		
	git diff
		visualizza differenze tra working directory e staging area
		
	git diff --staged
		visualizza differenze tra staging area e ultimo commit
		
git clone:
	copia una repository
	
	git clone URL
		copia da un url
		
git init:
	inizializza una repository (nella cartella corrente)
	
git status:
	verifica situazione attuale, files modificati e nella tracking area ecc ecc
	
git add:
	aggiunge file alla staging area
	
	git add <file>
		aggiunge quel file
		
git reset:
	rimuove modifiche
	
	git reset --hard
		elimina TUTTE le modifiche sia in working directory sia in staging area (PERDI TUTTE LE MODIFICHE!)
		
		
git config:
	configura info
	
	git config --global user.email "you@example.com"
		conf email
		
	git config --global user.name "Your Name"
		conf nome utente
		
git branch:
	crea/visualizza branch
	
	git branch
		visualizza branches
		
	git branch nome
		crea nuovo branch
		
	git branch nome commit
		crea nuovo branch con HEAD commit indicato
		
	git branch -d nome
		elimina branch con quel nome
		
git merge:
	fa il merge di due o più branches
	
	git merge xxx yyy
		merge di yyy dentro xxx
		
	git merge --abort
		annulla il procedimento
		
git show:
	
	git show commit1
		visualizza differenze tra un commit e suo padre
		
		
git remote:
	visualizza/crea remotes
	
	git remote
		visualizza remote
		
	git remote add nome url
		aggiunge un remote di nome "nome" di url "url"
		
	git remote -v
		visualizza remotes con informazioni
		
git push:
	da locale a online
	
	git push remote branch
		
git pull:
	da online a locale
	
	git pull remote branch
	
git fetch:
	preleva da online senza fare merge (crea nuovo branch se vi sono commit locali)
	
	git fetch remote
